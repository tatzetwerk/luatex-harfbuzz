\def\harfbuzzswig#1{%
	\directlua{%
		hb_location = "#1"%
		dofile(kpse.find_file("luatex-swiglib.lua","tex"))%
		hb = dofile(kpse.find_file("harfbuzz-swig.lua","tex"))%
		dofile(kpse.find_file("font-hb.lua","tex"))%
	}%
}%
\def\harfbuzzswigdj#1{%	alternative binding by means of "luaharfbuzz" (Lua bindings for Harfbuzz by Deepak Jois: https://github.com/deepakjois/luaharfbuzz)
	\directlua{%
		hb_location = "#1"%
		dofile(kpse.find_file("luatex-swiglib.lua","tex"))%
		hb = dofile(kpse.find_file("harfbuzz-swig-dj.lua","tex"))%
		dofile(kpse.find_file("font-hb.lua","tex"))%
	}%
}%
\def\harfbuzzffi#1{%	alternative binding without SwigLib by means of FFI (this requires luajittex instead of luatex)
	\directlua{%
		hb_location = "#1"%
		hb = dofile(kpse.find_file("harfbuzz-ffi.lua","tex"))%
		dofile(kpse.find_file("font-hb.lua","tex"))%
	}%
}%
